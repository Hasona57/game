const canvas = document.createElement("canvas"); const ctx = canvas.getContext("2d"); canvas.width = 800; canvas.height = 500; document.body.appendChild(canvas); type ObjectType = "trash" | "tree"; class GameObject { constructor(public x: number, public y: number, public type: ObjectType) {} draw() { ctx.fillStyle = this.type === "trash" ? "red" : "green"; ctx.beginPath(); ctx.arc(this.x, this.y, 15, 0, Math.PI * 2); ctx.fill(); } } let objects: GameObject[] = []; let score = 0; let timer = 60; function spawnObject() { const x = Math.random() * canvas.width; const y = Math.random() * canvas.height; const type = Math.random() > 0.5 ? "trash" : "tree"; objects.push(new GameObject(x, y, type)); } function updateGame() { ctx.clearRect(0, 0, canvas.width, canvas.height); objects.forEach(obj => obj.draw()); ctx.fillStyle = "black"; ctx.fillText(`Score: ${score}`, 10, 20); ctx.fillText(`Time: ${timer}`, 10, 40); } canvas.addEventListener("click", (e) => { objects = objects.filter(obj => { const dx = e.clientX - obj.x; const dy = e.clientY - obj.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance < 15) { if (obj.type === "trash") score += 10; return false; } return true; }); }); setInterval(() => { if (timer > 0) { spawnObject(); updateGame(); timer--; } else { ctx.fillText("Game Over!", canvas.width / 2 - 30, canvas.height / 2); } }, 1000);
